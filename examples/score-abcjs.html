      function hasAbc() {
        return typeof window !== 'undefined' && window.ABCJS && typeof window.ABCJS.renderAbc === 'function';
      }

      // Demo pieces
      function cMajorScale() {
        const notes = [];
        const scale = [60,62,64,65,67,69,71,72];
        for (let i = 0; i < scale.length; i++) {
          notes.push({ pitch: scale[i], duration: 1, time: i });
        }
        return {
          tempo: 120,
          timeSignature: '4/4',
          keySignature: 'C',
          metadata: { title: 'C-major scale' },
          tracks: [{ notes }]
        };
      }

      function articulationDemo() {
        return {
          tempo: 108,
          timeSignature: '4/4',
          keySignature: 'C',
          metadata: { title: 'Articulations & Slide' },
          tracks: [{
            notes: [
              { pitch: 60, duration: 1, time: 0, articulations: ['staccato','accent'] },
              { pitch: 64, duration: 1, time: 1, articulations: [{ type: 'glissando', target: 67 }] },
              { pitch: 67, duration: 1, time: 2, articulations: ['tenuto'] }
            ]
          }]
        };
      }

      const elScore = document.getElementById('score');

      function renderScore() {
        if (!jm) {
          elScore.innerHTML = '
JMON bundle not found.
';
          return;
        }
        elNote.textContent = hasAbc() ? 'ABCJS detected.' : 'ABCJS not detected (showing text if possible).';

        try {
          const notation = jm.converters.abc(piece, { showArticulations: true });
          elAbc.textContent = notation;

          // Clear previous drawing
          elScore.innerHTML = '';

          if (hasAbc()) {
            window.ABCJS.renderAbc(elScore, notation, {
              responsive: 'resize',
              scale: 0.95,
              staffwidth: elScore.clientWidth ? Math.min(960, elScore.clientWidth - 16) : 760
            });
          } else {
            // Fallback: show text if ABCJS missing
            const pre = document.createElement('pre');
            pre.style.margin = '0';
            pre.style.color = '#cfe3ff';
            pre.style.fontSize = '12px';
            pre.textContent = notation;
            elScore.appendChild(pre);
          }
        } catch (e) {
          console.error('[ABC] Render failed:', e);
          elScore.innerHTML = '
Failed to render ABC.
';
        }
      }

      // Wire controls
      elBtnRender.addEventListener('click', renderScore);
      elBtnScale.addEventListener('click', () => {
        piece = cMajorScale();
        renderScore();
      });
      elBtnArtic.addEventListener('click', () => {
        piece = articulationDemo();
        renderScore();
      });

      // Initial render after DOM ready
      if (document.readyState === 'complete' || document.readyState === 'interactive') {
        setTimeout(renderScore, 0);
      } else {
        document.addEventListener('DOMContentLoaded', renderScore, { once: true });
      }

      // Resize re-render
      window.addEventListener('resize', () => {
        // Only re-render if ABCJS is present; text fallback doesn't need reflow
        if (hasAbc()) renderScore();
      });
